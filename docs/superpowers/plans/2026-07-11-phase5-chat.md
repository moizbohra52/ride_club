# Phase 5 — Realtime Group Chat Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:executing-plans. Steps use checkbox (`- [ ]`) syntax.

**Goal:** Per-ride realtime group chat with text + location messages, a typing indicator, seen-by read receipts, and an unread badge on Ride Detail, Live Map, and My Rides cards.

**Architecture:** A `ChatService` (GetxService) over RTDB `chats/{rideId}/messages`, plus `lastRead/{rideId}/{uid}` and the existing `presence` node for typing. A `modules/chat/` screen renders bubbles (text + flutter_map location thumbnail), a composer, typing row, and seen-by. Unread streams feed badges into existing controllers.

**Tech Stack:** firebase_database, GetX, flutter_map/latlong2 (location bubble + full map), existing LocationService/UserService/RideService.

## Global Constraints

- RTDB paths: `chats/{rideId}/messages/{msgId}`, `lastRead/{rideId}/{uid}`, `presence/{rideId}/{uid}/typing`. Timestamps via `ServerValue.timestamp`.
- No Google Maps — location bubble uses flutter_map + `AppConstants.osmTileUrl`.
- Writes timeout-guarded; typed friendly errors via `UiHelpers`.
- `latlong2` `Path` clash → import `hide Path` where `dart:ui` Path is also used (chat bubble uses only LatLng/Marker, so safe; still import latlong2 plainly).
- `flutter analyze` clean; tests pass.

```dart
// lib/models/chat_message.dart
class ChatMessage {
  final String id;
  final String senderId;
  final String senderName;
  final String? senderPhoto;
  final String type; // 'text' | 'location'
  final String? text;
  final double? lat;
  final double? lng;
  final int timestamp; // epoch ms
  final Set<String> readBy;

  const ChatMessage({
    required this.id,
    required this.senderId,
    required this.senderName,
    this.senderPhoto,
    required this.type,
    this.text,
    this.lat,
    this.lng,
    required this.timestamp,
    this.readBy = const <String>{},
  });

  bool isMine(String uid) => senderId == uid;
  bool get isLocation => type == 'location';
  int get seenByCount => readBy.where((u) => u != senderId).length;

  String get timeText {
    final d = DateTime.fromMillisecondsSinceEpoch(timestamp);
    final h = d.hour % 12 == 0 ? 12 : d.hour % 12;
    final m = d.minute.toString().padLeft(2, '0');
    final ap = d.hour < 12 ? 'AM' : 'PM';
    return '$h:$m $ap';
  }

  static double? _d(dynamic v) => v == null ? null : (v as num).toDouble();

  factory ChatMessage.fromMap(String id, Map<dynamic, dynamic> m) {
    final rb = m['readBy'];
    final Set<String> read = rb is Map
        ? rb.entries.where((e) => e.value == true).map((e) => e.key.toString()).toSet()
        : <String>{};
    return ChatMessage(
      id: id,
      senderId: (m['senderId'] ?? '') as String,
      senderName: (m['senderName'] ?? '') as String,
      senderPhoto: m['senderPhoto'] as String?,
      type: (m['type'] ?? 'text') as String,
      text: m['text'] as String?,
      lat: _d(m['lat']),
      lng: _d(m['lng']),
      timestamp: (m['timestamp'] ?? 0) is num ? (m['timestamp'] as num).toInt() : 0,
      readBy: read,
    );
  }
}

int unreadCountFor(List<ChatMessage> msgs, int lastReadMs, String myUid) {
  return msgs
      .where((m) => m.timestamp > lastReadMs && m.senderId != myUid)
      .length;
}
```

```dart
// lib/services/chat_service.dart
import 'package:firebase_database/firebase_database.dart';
import 'package:get/get.dart';
import '../core/utils/logger.dart';
import '../models/chat_message.dart';
import 'auth_service.dart';
import 'user_service.dart';

class ChatService extends GetxService {
  final FirebaseDatabase _db = FirebaseDatabase.instance;
  final AuthService _auth = Get.find<AuthService>();
  final UserService _users = Get.find<UserService>();
  static const Duration _timeout = Duration(seconds: 15);

  DatabaseReference _messages(String rideId) => _db.ref('chats/$rideId/messages');
  DatabaseReference _lastRead(String rideId, String uid) =>
      _db.ref('lastRead/$rideId/$uid');
  DatabaseReference _typing(String rideId, String uid) =>
      _db.ref('presence/$rideId/$uid/typing');

  Stream<List<ChatMessage>> watchMessages(String rideId) {
    return _messages(rideId).limitToLast(200).onValue.map((event) {
      final raw = event.snapshot.value as Map<dynamic, dynamic>?;
      if (raw == null) return <ChatMessage>[];
      final list = raw.entries
          .map((e) => ChatMessage.fromMap(
              e.key as String, e.value as Map<dynamic, dynamic>))
          .toList()
        ..sort((a, b) => a.timestamp.compareTo(b.timestamp));
      return list;
    });
  }

  Future<void> sendText(String rideId, String text) async {
    final t = text.trim();
    if (t.isEmpty) return;
    await _push(rideId, {'type': 'text', 'text': t});
  }

  Future<void> sendLocation(String rideId, double lat, double lng) async {
    await _push(rideId, {'type': 'location', 'lat': lat, 'lng': lng});
  }

  Future<void> _push(String rideId, Map<String, dynamic> body) async {
    final uid = _auth.uid;
    if (uid == null) return;
    final profile = await _users.fetch(uid);
    final data = <String, dynamic>{
      'senderId': uid,
      'senderName': profile?.name ?? 'Rider',
      'senderPhoto': profile?.photoUrl,
      'timestamp': ServerValue.timestamp,
      'readBy': <String, dynamic>{uid: true},
      ...body,
    };
    await _messages(rideId).push().set(data).timeout(_timeout,
        onTimeout: () => throw Exception("Couldn't send. Check your connection."));
  }

  Future<void> markRead(String rideId, List<ChatMessage> visible) async {
    final uid = _auth.uid;
    if (uid == null) return;
    await _lastRead(rideId, uid).set(ServerValue.timestamp);
    for (final m in visible) {
      if (m.senderId == uid) continue;
      if (m.readBy.contains(uid)) continue;
      _messages(rideId).child(m.id).child('readBy').child(uid).set(true).catchError(
          (Object e) => Log.e('readBy write failed', error: e));
    }
  }

  void setTyping(String rideId, bool typing) {
    final uid = _auth.uid;
    if (uid == null) return;
    _typing(rideId, uid).set(typing);
  }

  Stream<List<String>> watchTyping(String rideId) {
    final uid = _auth.uid;
    return _db.ref('presence/$rideId').onValue.map((event) {
      final raw = event.snapshot.value as Map<dynamic, dynamic>?;
      if (raw == null) return <String>[];
      final typers = <String>[];
      raw.forEach((key, value) {
        if (key == uid) return;
        if (value is Map && value['typing'] == true) typers.add(key as String);
      });
      return typers;
    });
  }

  Stream<int> unreadCount(String rideId) {
    final uid = _auth.uid;
    if (uid == null) return Stream<int>.value(0);
    // Combine messages with lastRead; re-emit on either change.
    return _messages(rideId).limitToLast(200).onValue.asyncMap((event) async {
      final raw = event.snapshot.value as Map<dynamic, dynamic>?;
      final msgs = raw == null
          ? <ChatMessage>[]
          : raw.entries
              .map((e) => ChatMessage.fromMap(
                  e.key as String, e.value as Map<dynamic, dynamic>))
              .toList();
      final lrSnap = await _lastRead(rideId, uid).get();
      final lastRead = (lrSnap.value as num?)?.toInt() ?? 0;
      // First-ever open (no lastRead): treat all as read to avoid a huge count.
      if (lastRead == 0) return 0;
      return unreadCountFor(msgs, lastRead, uid);
    });
  }
}
```

```dart
// lib/modules/chat/chat_controller.dart
import 'dart:async';
import 'package:flutter/widgets.dart';
import 'package:get/get.dart';
import '../../core/utils/ui_helpers.dart';
import '../../models/chat_message.dart';
import '../../services/auth_service.dart';
import '../../services/chat_service.dart';
import '../../services/location_service.dart';
import '../../services/ride_service.dart';

class ChatController extends GetxController {
  final ChatService _chat = Get.find<ChatService>();
  final RideService _rides = Get.find<RideService>();
  final LocationService _loc = Get.find<LocationService>();
  final AuthService _auth = Get.find<AuthService>();
  final String rideId;
  ChatController(this.rideId);

  final RxList<ChatMessage> messages = <ChatMessage>[].obs;
  final RxList<String> typingUids = <String>[].obs;
  final RxInt memberCount = 0.obs;
  final RxBool sending = false.obs;
  final TextEditingController input = TextEditingController();
  Timer? _typingTimer;

  String? get uid => _auth.uid;

  @override
  void onInit() {
    super.onInit();
    messages.bindStream(_chat.watchMessages(rideId));
    typingUids.bindStream(_chat.watchTyping(rideId));
    _rides.watchMembers(rideId).listen((m) => memberCount.value = m.length);
    // mark read whenever messages change while we're on screen
    ever<List<ChatMessage>>(messages, (list) => _chat.markRead(rideId, list));
  }

  void onChanged(String _) {
    _chat.setTyping(rideId, true);
    _typingTimer?.cancel();
    _typingTimer = Timer(const Duration(seconds: 2),
        () => _chat.setTyping(rideId, false));
  }

  Future<void> send() async {
    final text = input.text;
    if (text.trim().isEmpty) return;
    input.clear();
    _chat.setTyping(rideId, false);
    try {
      await _chat.sendText(rideId, text);
    } catch (e) {
      input.text = text; // restore so the user can retry
      UiHelpers.error(e.toString().replaceFirst('Exception: ', ''));
    }
  }

  Future<void> sendMyLocation() async {
    sending.value = true;
    try {
      final res = await _loc.ensurePermission();
      if (res != LocationPermissionResult.granted) {
        UiHelpers.error('Location permission is needed to share your spot.');
        return;
      }
      final pos = await _loc.currentPosition();
      if (pos == null) {
        UiHelpers.error('Could not get your location.');
        return;
      }
      await _chat.sendLocation(rideId, pos.latitude, pos.longitude);
    } catch (e) {
      UiHelpers.error(e.toString().replaceFirst('Exception: ', ''));
    } finally {
      sending.value = false;
    }
  }

  @override
  void onClose() {
    _typingTimer?.cancel();
    _chat.setTyping(rideId, false);
    input.dispose();
    super.onClose();
  }
}
```

```dart
class _Bubble extends StatelessWidget {
  final ChatMessage msg;
  final bool mine;
  final int memberCount;
  const _Bubble({required this.msg, required this.mine, required this.memberCount});

  @override
  Widget build(BuildContext context) {
    final scheme = Theme.of(context).colorScheme;
    final bg = mine ? AppColors.seed : scheme.surfaceContainerHighest;
    final fg = mine ? Colors.white : scheme.onSurface;
    return Column(
      crossAxisAlignment: mine ? CrossAxisAlignment.end : CrossAxisAlignment.start,

                style: TextStyle(fontSize: 11, color: scheme.onSurfaceVariant)),

          constraints: const BoxConstraints(maxWidth: 280),
          decoration: BoxDecoration(
            color: bg,
            borderRadius: BorderRadius.circular(16),
          ),
          clipBehavior: Clip.antiAlias,
          child: msg.isLocation

              : Padding(
                  padding: const EdgeInsets.symmetric(horizontal: 14, vertical: 10),
                  child: Text(msg.text ?? '', style: TextStyle(color: fg)),
                ),
        ),
        Padding(
          padding: const EdgeInsets.only(left: 8, right: 8, bottom: 6),
          child: Text(
            mine ? '${msg.timeText} · ${_seen()}' : msg.timeText,
            style: TextStyle(fontSize: 10, color: scheme.onSurfaceVariant),
          ),
        ),
      ],
    );
  }

  String _seen() {
    final n = msg.seenByCount;
    if (n <= 0) return 'Sent';
    if (memberCount > 1 && n >= memberCount - 1) return 'Seen';

                    initialCenter: point, initialZoom: 15,

                        flags: InteractiveFlag.none)),

                    urlTemplate: AppConstants.osmTileUrl,
```
